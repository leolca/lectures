    \canto@fil$\m@th \CancelColor

      \multiply\dimen@#4\divide\dimen@#3 \vrule\@depth\dimen@\@width\z@
      \vector(#3,-#4){#5}%
    }}_{\raise-#2\dimen@\copy\z@\kern-\scriptspace}$%

    \hfil \box\@tempboxa \kern\wd\z@ \hfil \cr}}
